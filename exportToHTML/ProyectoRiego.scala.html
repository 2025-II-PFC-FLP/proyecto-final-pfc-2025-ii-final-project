package proyecto.riego

import scala.util.Random
import scala.collection.immutable.Vector

object ProyectoRiego {

  // Tipos
  type Tablon = (Int, Int, Int)           // (tsup, treg, prio)
  type Finca = Vector[Tablon]
  type Distancia = Vector[Vector[Int]]
  type ProgRiego = Vector[Int]
  type TiempoInicioRiego = Vector[Int]

  private val random = new Random()

  // ----------- Generación aleatoria -----------

  def fincaAlAzar(long: Int): Finca = {
    Vector.fill(long)(
      (random.nextInt(long * 2) + 1,
        random.nextInt(long) + 1,
        random.nextInt(4) + 1)
    )
  }

  def distanciaAlAzar(long: Int): Distancia = {
    val raw = Vector.fill(long, long)(random.nextInt(long * 3) + 1)
    Vector.tabulate(long, long) { (i, j) =>
      if (i < j) raw(i)(j)
      else if (i == j) 0
      else raw(j)(i)
    }
  }

  // ----------- Exploración -----------

  def tsup(f: Finca, i: Int): Int = f(i)._1
  def treg(f: Finca, i: Int): Int = f(i)._2
  def prio(f: Finca, i: Int): Int = f(i)._3

  // ----------- tIR -----------

  def tIR(f: Finca, pi: ProgRiego): TiempoInicioRiego = {
    val n = f.length
    val permTurno = Vector.tabulate(n)(j => pi.indexOf(j))

    def acum(j: Int): Vector[Int] = {
      if (j == 0) Vector(0)
      else {
        val prev = acum(j - 1)
        val prevStart = prev.last
        val prevTablon = permTurno(j - 1)
        prev :+ (prevStart + treg(f, prevTablon))
      }
    }

    val starts = acum(n - 1)
    Vector.tabulate(n)(i => starts(pi(i)))
  }

  // ----------- Costos -----------

  def costoRiegoTablon(i: Int, f: Finca, pi: ProgRiego): Int = {
    val t = tIR(f, pi)(i)
    val tsi = tsup(f, i)
    val tri = treg(f, i)
    val pr = prio(f, i)

    if (tsi - tri >= t)
      tsi - (t + tri)
    else
      pr * ((t + tri) - tsi)
  }

  def costoRiegoFinca(f: Finca, pi: ProgRiego): Int = {
    def aux(i: Int): Int =
      if (i >= f.length) 0
      else costoRiegoTablon(i, f, pi) + aux(i + 1)
    aux(0)
  }
